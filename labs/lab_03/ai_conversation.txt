def create_transaction(transactions, prices_dates, date, trn_type, ticker=None, shares=None, cash_amount=None):
    """
    Creates a transaction and appends it to the transactions list.
    Enforces:
    - prices_dates.json as source of truth
    - automatic price lookup
    - cash + stock legs
    - sorting by (date, ticker) with $$$$ first
    - IDs reflect final time order
    """

    # --- 1. Validate date ---
    if date not in prices_dates:
        print("Invalid date")
        return None

    # --- 2. Handle transaction types ---
    if trn_type in ['BUY', 'SELL']:

        # Validate ticker
        valid_tickers = [row['ticker'] for row in prices_dates[date]]
        if ticker not in valid_tickers:
            print("Invalid ticker for that date")
            return None

        # Lookup price
        price = None
        for row in prices_dates[date]:
            if row['ticker'] == ticker:
                price = row['raw_price']
                break

        if price is None:
            print("Price not found")
            return None

        # Compute cash impact
        total = shares * price
        if trn_type == 'BUY':
            cash_flow = -total
        else:  # SELL
            cash_flow = total

        # --- Stock transaction ---
        stock_trn = {
            'id': None,
            'date': date,
            'trn_type': trn_type,
            'ticker': ticker,
            'shares': shares if trn_type == 'BUY' else -shares,
            'price': price,
            'cash_amount': None
        }

        # --- Cash transaction ---
        cash_trn = {
            'id': None,
            'date': date,
            'trn_type': trn_type,
            'ticker': '$$$$',
            'shares': None,
            'price': 1.0,
            'cash_amount': cash_flow
        }

        transactions.append(stock_trn)
        transactions.append(cash_trn)

    elif trn_type in ['CONTRIBUTION', 'WITHDRAWAL']:

        if trn_type == 'CONTRIBUTION':
            cash_flow = cash_amount
        else:
            cash_flow = -cash_amount

        cash_trn = {
            'id': None,
            'date': date,
            'trn_type': trn_type,
            'ticker': '$$$$',
            'shares': None,
            'price': 1.0,
            'cash_amount': cash_flow
        }

        transactions.append(cash_trn)

    else:
        print("Invalid transaction type")
        return None

    # --- 3. Sort transactions ---
    def sort_key(trn):
        ticker_key = '' if trn['ticker'] == '$$$$' else trn['ticker']
        return (trn['date'], ticker_key)

    transactions.sort(key=sort_key)

    # --- 4. Reassign IDs (time-ordered) ---
    for i, trn in enumerate(transactions, start=1):
        trn['id'] = i

    return None

def get_cash_balance(transactions, as_of_date):
    """
    Returns the cash balance as of a given date.
    Cash is reconstructed by summing all $$$$ transactions up to that date.
    """

    cash = 0.0

    for trn in transactions:

        # since transactions are sorted, we can stop early
        if trn['date'] > as_of_date:
            break

        if trn['ticker'] == '$$$$':
            cash += trn['cash_amount']

    return cash

def build_portfolio(transactions, as_of_date):
    """
    Reconstructs portfolio holdings (shares and cost) as of a given date.
    Uses transaction history (excluding $$$$) up to that date.
    """

    portfolio = {}

    for trn in transactions:

        # stop early (transactions are sorted)
        if trn['date'] > as_of_date:
            break

        ticker = trn['ticker']

        # skip cash
        if ticker == '$$$$':
            continue

        # initialize ticker if not present
        if ticker not in portfolio:
            portfolio[ticker] = {
                'shares': 0,
                'cost': 0.0
            }

        # handle BUY
        if trn['trn_type'] == 'BUY':
            portfolio[ticker]['shares'] += trn['shares']
            portfolio[ticker]['cost'] += trn['shares'] * trn['price']

        # handle SELL (FIFO not implemented — simple reduction)
        elif trn['trn_type'] == 'SELL':
            portfolio[ticker]['shares'] += trn['shares']  # shares already negative

            # simple proportional cost reduction
            if portfolio[ticker]['shares'] != 0:
                avg_cost = portfolio[ticker]['cost'] / (portfolio[ticker]['shares'] - trn['shares'])
                portfolio[ticker]['cost'] += trn['shares'] * avg_cost
            else:
                portfolio[ticker]['cost'] = 0.0

    return portfolio

value_portfolio (final)
def value_portfolio(transactions, prices_dates, as_of_date):
    """
    Returns a full valuation of the portfolio as of a given date.
    Includes cash, per-position values, and total value.
    """

    # --- 1. Validate date ---
    if as_of_date not in prices_dates:
        print("Invalid valuation date")
        return None

    # --- 2. Get cash and positions ---
    cash = get_cash_balance(transactions, as_of_date)
    portfolio = build_portfolio(transactions, as_of_date)

    # --- 3. Build price lookup for the date ---
    price_map = {}
    for row in prices_dates[as_of_date]:
        price_map[row['ticker']] = row['raw_price']

    # --- 4. Value positions ---
    positions_value = {}
    total_positions_value = 0.0

    for ticker, data in portfolio.items():

        shares = data['shares']

        # skip if no shares (optional but clean)
        if shares == 0:
            continue

        if ticker not in price_map:
            print(f"Missing price for {ticker} on {as_of_date}")
            return None

        price = price_map[ticker]
        market_value = shares * price

        positions_value[ticker] = {
            'shares': shares,
            'price': price,
            'market_value': market_value
        }

        total_positions_value += market_value

    # --- 5. Compute total value ---
    total_value = cash + total_positions_value

    # --- 6. Return structured output ---
    return {
        'cash': cash,
        'positions': positions_value,
        'total_value': total_value
    }

list_transactions_for_ticker (final)
def list_transactions_for_ticker(transactions, ticker, start_date=None, end_date=None, trn_type=None):
    """
    Returns transactions for a given ticker with optional:
    - date filtering (start_date, end_date)
    - transaction type filtering (trn_type)
    - grouping of stock + cash legs (by date + type)
    """

    result = []

    # --- 1. filter transactions ---
    for trn in transactions:

        # ticker filter
        if trn['ticker'] != ticker and trn['ticker'] != '$$$$':
            continue

        # transaction type filter
        if trn_type is not None and trn['trn_type'] != trn_type:
            continue

        # date filters
        if start_date is not None and trn['date'] < start_date:
            continue

        if end_date is not None and trn['date'] > end_date:
            continue

        result.append(trn)

    # --- 2. group stock + cash legs ---
    grouped = {}

    for trn in result:

        key = (trn['date'], trn['trn_type'])

        if key not in grouped:
            grouped[key] = {
                'date': trn['date'],
                'trn_type': trn['trn_type'],
                'ticker': None,
                'shares': None,
                'price': None,
                'cash_amount': 0.0
            }

        # assign stock side
        if trn['ticker'] != '$$$$':
            grouped[key]['ticker'] = trn['ticker']
            grouped[key]['shares'] = trn.get('shares')
            grouped[key]['price'] = trn.get('price')

        # accumulate cash side
        if trn['ticker'] == '$$$$':
            grouped[key]['cash_amount'] += trn.get('cash_amount', 0.0)

    # --- 3. convert grouped dict to sorted list ---
    output = []

    for key in sorted(grouped.keys()):
        output.append(grouped[key])

    return output
